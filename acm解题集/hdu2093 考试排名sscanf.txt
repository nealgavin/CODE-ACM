#include<stdio.h>
#include<cstring>
#include<algorithm>
#define N 1003
using namespace std;
struct ac
{
  char name[20];
  int num;
  int time;
};
ac stu[N];
bool cmp(ac a,ac b)
{
  if(a.num==b.num&&a.time==b.time)return strcmp(a.name,b.name)<0;
  if(a.num==b.num)return a.time<b.time;
  return a.num>b.num;
}
int main()
{
  int n,m,i,j,res,tim,num;//NÌâÊý£¬M·£Ê±
  char type[20];
  //freopen("1.txt","r",stdin);
  scanf("%d%d",&n,&m);
  for(i=0;i<=1002;i++)
  {
    stu[i].num=stu[i].time=0;
  }
  for(i=0;scanf("%s",stu[i].name)!=EOF;i++)
  {
    for(j=0;j<n;j++)
    {
      scanf("%s",type);
      res=sscanf(type,"%d(%d)",&tim,&num);
      if(res==2)
      {
        stu[i].num++;
        stu[i].time+=tim+num*m;
      }
      if(res==1&&tim>0)
      {
        stu[i].num++;
        stu[i].time+=tim;
      }
    }

  }
  sort(stu,stu+i,cmp);
  for(j=0;j<i;j++)
  printf("%-10s %2d %4d\n",stu[j].name,stu[j].num,stu[j].time);
}
